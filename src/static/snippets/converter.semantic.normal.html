// Offers methods to convert text between camelcase (z.B. textInNotation) and underscore (z.B. text_in_notation) notation
public class UnderScoreCamelCaseConverter {

	public String convertUnderScoreToCamelCase(String underScore) {
		String[] singleWordParts = underScore.split("_");
		String convertedInput = singleWordParts[0];
		for (int i = 1; i < singleWordParts.length; i++) {
			convertedInput += lowercaseFirstLetter(singleWordParts[i]);
		}
		return convertedInput;
	}

	private String capitalizeFirstLetter(String notConverted) {
		String firstLetter = notConverted.substring(0, 1);
		String otherLetters = notConverted.substring(1);
		String convertedWord = firstLetter.toUpperCase().concat(otherLetters);
		return convertedWord;
	}

	private String lowercaseFirstLetter(String notConverted) {
		String firstLetter = notConverted.substring(0, 1);
		String otherLetters = notConverted.substring(1);
		String convertedWord = firstLetter.toLowerCase().concat(otherLetters);
		return convertedWord;
	}
}
