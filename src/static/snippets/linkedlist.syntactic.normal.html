// content of listelements must not be null

	Entry<T> head;

		Entry<T> sentinel = new Entry<>();
		sentinel.next = head;
		while (currentEntry.next != null) {
			if (currentEntry.next.content.equals(searched)) {
				currentEntry.deleteNextEntry()
			} else {
				currentEntry = currentEntry.next;
			}
		}
		this.head = sentinel.next;
	}

	private static class Entry<T> {
		Entry<T> next;
		T content;

		public void deleteNextEntry() {
			next = next.next;
		}
	}
}
